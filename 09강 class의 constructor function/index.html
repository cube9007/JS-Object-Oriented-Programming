09강 class의 constructor function

코드 예시

class Person { constructor(name, first, second) { this.name = name; this.first = first; this.second = second; } } var kim = new Person('kim', 10, 20); document.write('kim.name : ' + kim.name + '<br>'); document.write('kim.first : ' + kim.first + '<br>'); document.write('kim.second : ' + kim.second + '<br>');

기존 function 코드

function Person2(name, first, second) { this.name = name; this.first = first; this.second = second; } var kim = new Person2('kim', 10, 20); document.write( 'kim.name : ' + kim.name + '<br>'); document.write( 'kim.first : ' + kim.first + '<br>'); document.write( 'kim.second : ' + kim.second + '<br>');
