생성자 new 

객체를 그냥 쓰면 객체이지만
객체 앞에다 new를 붙이면 생성자가 됨
function Person(name, first, second, third) { this.name = name; this.first = first; this.second = second; this.third = third ; this.sum = function() { return this.first + this.second + this.third; } } var wonho = new Person('choiwonho', 10, 20, 30); var yuri = new Person('kimyuri', 10, 30, 50); var daegeun = new Person('jangdaegeun', 20, 10, 30); document.write(wonho.sum()+'<br>'); document.write(yuri.sum()+'<br>'); document.write(daegeun.sum()+'<br>');
